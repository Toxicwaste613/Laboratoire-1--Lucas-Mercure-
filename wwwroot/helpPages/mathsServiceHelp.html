Service /api/maths - Documentation

Help page:

List of possible query string:



____________________________________________________________


? op = + & x = number & y = number

return {"op":"+","x":number,"y":number,"value":x+y}


? op = - & x = number & y = number

return {"op":"-","x":number,"y":number,"value":x-y}


? op = * & x = number & y = number

return {"op":"*","x":number,"y":number,"value":x*y}


? op = / & x = number & y = number

return {"op":"/","x":number,"y":number,"value":x/y}


? op = % & x = number & y = number

return {"op":"%","x":number,"y":number,"value":x%y}


? op = ! & x = number & y = number

return {"op":"%","n":"integer","value":n!}


? op = p & x = number & y = number

return {"op":"p","n":integer, "value":true if n is prime number}


? op = np & x = number & y = number

return {"op":"n","n":integer,"value":nth prime number}
